Droapps UI . Custom Select

by Diego Tres

Description

Custom Select is a skin for select form tag

Dependencies

  • jQuery 1.4.2+
  • jQuery-UI 1.8.2+

Demo

$('#demo select').customselect();

Options

visible /// Define the amount of visible items.
possible values: any number or 'auto'
default: 'auto'
position /// Define position to open the content.
possible values: bottom | down / top | up / middle | center
default: 'bottom'
duration /// Define the duration of animation. ( open / close )
possible values: any number
default: 200
easing /// Define the easing of animation. ( open / close )
possible values: any easing name present at jquery.ui.effects.core
default: 'linear'
classname /// Add a class to main element of plugin ( dl )
possible values: any string
default: 'd3'
context /// scope to execute the callback functions
possible values: any scope variable
default: null

Events

beforeOpen /// function to execute before animation to open start
possible values: an function
default: null
onOpen /// function to execute when animation to open is finished
possible values: an function
default: null
beforeClose /// function to execute before animation to close start
possible values: an function
default: null
onClose /// function to execute when animation to close is finished
possible values: an function
default: null
change /// function that return ( event , value ) after an item was selected
possible values: an function
.customselect({ change: function( event , value ) {} })

Methods

destroy /// remove the element and clean instance
.customselect('destroy')
disable /// disable customselect
.customselect('disable')
enable /// enable customselect
.customselect('enable')
options /// get or set any autocomplete option. If no value is specified, will act as a getter.
.customselect( 'option' , optionName , [value] )
value /// get or set value
.customselect( 'value' , [new_value] )
widget /// return the customselect element
.customselect('widget')
